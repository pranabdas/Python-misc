Create meshgrid:
<Tabs
  defaultValue="input"
  values={[
    { label: 'Input', value: 'input', },
    { label: 'Output', value: 'output', },
  ]
}>

<TabItem value="input">

```python
x = np.array([-1, 0, 1])
y = np.array([-2, 0, 2])

print(X, '\n\n', Y)
```

</TabItem>

<TabItem value="output">

```python

 [-1  0  1]
 [-1  0  1]]

 [[-2 -2 -2]
 [ 0  0  0]
 [ 2  2  2]]
```

</TabItem>
</Tabs>
